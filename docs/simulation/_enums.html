  6class Direction(Enum):
  7    """Cardinal directions in the simulation grid, aligned with screen coordinates.
  8
  9    In this convention +x is right, +y is down, matching Canvas pixel space.
 10    This eliminates y-flip in the renderer and makes all coordinate math consistent.
 11
 12    Attributes:
 13        UP: (0, -1) — negative y direction.
 14        DOWN: (0, +1) — positive y direction.
 15        LEFT: (-1, 0) — negative x direction.
 16        RIGHT: (+1, 0) — positive x direction.
 17    """
 18    UP    = (0, -1)
 19    DOWN  = (0, +1)
 20    LEFT  = (-1, 0)
 21    RIGHT = (+1, 0)
 22
 23    def __matmul__(self, other):
 24        """Rotate this direction by a Rotation.
 25
 26        Args:
 27            other: A Rotation to apply.
 28
 29        Returns:
 30            The resulting Direction after rotation.
 31
 32        Raises:
 33            KeyError: If the rotated vector does not match a valid Direction.
 34        """
 35        from .utils import Vec
 36        new_direction = Vec(*self.value) @ other
 37        match new_direction.x, new_direction.y:
 38            case 0, -1:
 39                return Direction.UP
 40            case 0, 1:
 41                return Direction.DOWN
 42            case -1, 0:
 43                return Direction.LEFT
 44            case 1, 0:
 45                return Direction.RIGHT
 46            case _:
 47                raise KeyError(new_direction)

 50    """Clockwise rotations in 90-degree increments.
 51
 52    Attributes:
 53        ROT_0: No rotation (0°).
 54        ROT_1: 90° clockwise.
 55        ROT_2: 180° clockwise.
 56        ROT_3: 270° clockwise.
 57    """
 58    ROT_0 = auto()
 59    ROT_1 = auto()
 60    ROT_2 = auto()
 61    ROT_3 = auto()
 62    
 63    def __matmul__(self, other):
 64        """Compose two rotations.
 65
 66        Args:
 67            other: Another Rotation to compose with.
 68
 69        Returns:
 70            The resulting Rotation after composition.
 71        """
 72        match (self.value, other):
 73            case Rotation.ROT_0, _:
 74                return other
 75            case _, Rotation.ROT_0:
 76                return self
 77            case Rotation.ROT_1, Rotation.ROT_1:
 78                return Rotation.ROT_2
 79            case Rotation.ROT_1, Rotation.ROT_2:
 80                return Rotation.ROT_3
 81            case Rotation.ROT_1, Rotation.ROT_3:
 82                return Rotation.ROT_0
 83            case Rotation.ROT_2, Rotation.ROT_1:
 84                return Rotation.ROT_3
 85            case Rotation.ROT_2, Rotation.ROT_2:
 86                return Rotation.ROT_0
 87            case Rotation.ROT_2, Rotation.ROT_3:
 88                return Rotation.ROT_1
 89            case Rotation.ROT_3, Rotation.ROT_1:
 90                return Rotation.ROT_0
 91            case Rotation.ROT_3, Rotation.ROT_2:
 92                return Rotation.ROT_1
 93            case Rotation.ROT_3, Rotation.ROT_3:
 94                return Rotation.ROT_2
